#include <stdio.h>

int count_balanced_subarrays(int arr[], int n, int k) {
    int halfSize = k / 2;
    if (halfSize == 0) {
        return n;
    }

    int balancedCount = 0;
    long long leftSum = 0;
    long long rightSum = 0;

    for (int i = 0; i < halfSize; i++) {
        leftSum += arr[i];
    }

    int rightStartOffset = k - halfSize;
    for (int i = 0; i < halfSize; i++) {
        rightSum += arr[rightStartOffset + i];
    }

    if (leftSum == rightSum) {
        balancedCount++;
    }

    for (int i = 1; i <= n - k; i++) {
        leftSum = leftSum - arr[i - 1] + arr[i + halfSize - 1];
        rightSum = rightSum - arr[i + rightStartOffset - 1] + arr[i + k - 1];

        if (leftSum == rightSum) {
            balancedCount++;
        }
    }

    return balancedCount;
}

int main() {
    int n, k;
    scanf("%d %d", &n, &k);
    int arr[n];
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }
    
    int result = count_balanced_subarrays(arr, n, k);
    printf("%d\n", result);
    return 0;
}
