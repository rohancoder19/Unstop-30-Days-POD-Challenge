// Function declaration
int peakIndexInMountainArray(vector<int>& A);

int main() {
    // Optimize standard I/O streams for performance
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    if (cin >> n) {
        vector<int> arr(n);
        for (int i = 0; i < n; i++) {
            cin >> arr[i];
        }
        
        // Output the result returned by user function
        cout << peakIndexInMountainArray(arr) << endl;
    }
    return 0;
}

// Function definition (filled by the user)
int peakIndexInMountainArray(vector<int>& A) {
    int low = 0;
    int high = A.size() - 1;
    
    while (low < high) {
        int mid = low + (high - low) / 2;
        
        // If the element to the right is larger, the peak is on the right side
        if (A[mid] < A[mid + 1]) {
            low = mid + 1;
        } 
        // Otherwise, the peak is either at 'mid' or to its left side
        else {
            high = mid;
        }
    }
    
    // 'low' and 'high' converge to the peak index
    return low;
}
