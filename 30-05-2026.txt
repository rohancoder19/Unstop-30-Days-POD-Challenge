//C++

#include <iostream>
#include <vector>
#include <utility>
#include <queue>

using namespace std;

int max_collapsing_realities(int N, int M, int K, const vector<int>& unstable_realities, const vector<pair<int, int>>& dependencies) {
    vector<vector<int>> graph(N);
    vector<int> remaining_dependencies(N, 0);

    for (const auto& edge : dependencies) {
        int u = edge.first;
        int v = edge.second;
        graph[v].push_back(u);
        remaining_dependencies[u]++;
    }

    queue<int> ready_to_collapse;
    vector<bool> visited(N, false);

    for (int reality : unstable_realities) {
        if (remaining_dependencies[reality] == 0) {
            ready_to_collapse.push(reality);
            visited[reality] = true;
        }
    }

    int collapsed_count = 0;

    while (!ready_to_collapse.empty()) {
        int current = ready_to_collapse.front();
        ready_to_collapse.pop();
        collapsed_count++;

        for (int neighbor : graph[current]) {
            remaining_dependencies[neighbor]--;
            if (remaining_dependencies[neighbor] == 0 && !visited[neighbor]) {
                visited[neighbor] = true;
                ready_to_collapse.push(neighbor);
            }
        }
    }

    return collapsed_count;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, M;
    if (!(cin >> N >> M)) return 0;

    int K;
    cin >> K;

    vector<int> unstable_realities(K);
    for (int i = 0; i < K; i++) {
        cin >> unstable_realities[i];
    }

    vector<pair<int, int>> dependencies(M);
    for (int i = 0; i < M; i++) {
        cin >> dependencies[i].first >> dependencies[i].second;
    }

    cout << max_collapsing_realities(N, M, K, unstable_realities, dependencies) << "\n";

    return 0;
}
