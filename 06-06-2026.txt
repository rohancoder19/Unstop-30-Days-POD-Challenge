//C++

#include <iostream>
#include <string>

// Function where user will implement the logic
void decode_message(const std::string& S) {
    std::string decoded_str = "";
    int i = 0;
    int n = S.length();

    while (i < n) {
        // Look ahead 2 characters to see if there is a '#'
        if (i + 2 < n && S[i + 2] == '#') {
            // Extract the two-digit number
            int value = std::stoi(S.substr(i, 2));
            decoded_str += (char)('a' + value - 1);
            i += 3; // Move past the two digits and the '#'
        } else {
            // Extract the single-digit number
            int value = S[i] - '0';
            decoded_str += (char)('a' + value - 1);
            i += 1; // Move past the single digit
        }
    }

    // Output the final decoded message string followed by a newline
    std::cout << decoded_str << "\n";
}

int main() {
    // Optimize standard I/O streams for quick competitive programming execution
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    std::string S;
    if (std::getline(std::cin, S)) {
        // Call user logic function and print the output
        decode_message(S);
    }
    return 0;
}
