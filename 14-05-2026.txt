//C++

#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

struct RaidResult {
    int original_index;
    int stamina_used;
};

vector<int> evaluate_raids(int n, vector<string>& raids, vector<int>& stamina_limits) {
    vector<RaidResult> valid_raids;

    for (int i = 0; i < n; ++i) {
        string current_raid = raids[i];
        int limit = stamina_limits[i];
        
        if (current_raid.empty() || current_raid.back() != 'R') {
            continue;
        }

        bool has_touch = false;
        bool has_double_retreat = false;
        int current_used = 0;
        bool stamina_exceeded = false;

        for (int j = 0; j < current_raid.length(); ++j) {
            char move = current_raid[j];
            
            if (move == 'T') {
                current_used += 3;
                has_touch = true;
            } else if (move == 'F') {
                current_used += 1;
            } else if (move == 'R') {
                current_used += 2;
                if (j > 0 && current_raid[j-1] == 'R') {
                    has_double_retreat = true;
                    break;
                }
            }

            if (current_used > limit) {
                stamina_exceeded = true;
                break;
            }
        }

        if (!stamina_exceeded && !has_double_retreat && has_touch) {
            valid_raids.push_back({i, current_used});
        }
    }

    stable_sort(valid_raids.begin(), valid_raids.end(), [](const RaidResult& a, const RaidResult& b) {
        return a.stamina_used < b.stamina_used;
    });

    vector<int> final_indices;
    for (const auto& entry : valid_raids) {
        final_indices.push_back(entry.original_index);
    }

    return final_indices;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    if (!(cin >> n)) return 0;
    vector<string> raids(n);
    for (int i = 0; i < n; ++i) {
        cin >> raids[i];
    }
    vector<int> stamina(n);
    for (int i = 0; i < n; ++i) {
        cin >> stamina[i];
    }

    vector<int> result = evaluate_raids(n, raids, stamina);

    for (size_t i = 0; i < result.size(); ++i) {
        cout << result[i] << (i == result.size() - 1 ? "" : " ");
    }
    cout << endl;

    return 0;
}
